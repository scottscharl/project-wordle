export default function Guess({ word }) {
  const letters = [];
  for (let i = 0; i < word.length; i++) {
    letters.push(word[i]);
  }
  return (
    <div key={Math.random()} className="guess">
      {letters.map((letter) => (
        <div key={Math.random()} className="cell">
          {letter}
        </div>
      ))}
    </div>
  );
}
